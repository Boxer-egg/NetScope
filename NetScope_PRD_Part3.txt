8.4 ConnectionPoller 实现要点
核心逻辑
class ConnectionPoller {
    private var timer: DispatchSourceTimer?
    private let queue = DispatchQueue(label: "com.netscope.poller", qos: .utility)

    func start(interval: TimeInterval = 1.0, onUpdate: @escaping ([Connection]) -> Void) {
        timer = DispatchSource.makeTimerSource(queue: queue)
        timer?.schedule(deadline: .now(), repeating: interval)
        timer?.setEventHandler { [weak self] in
            let connections = self?.poll() ?? []
            DispatchQueue.main.async { onUpdate(connections) }
        }
        timer?.resume()
    }

    private func poll() -> [Connection] {
        let output = shell("/usr/sbin/lsof -nP -iTCP -iUDP -sTCP:ESTABLISHED,LISTEN")
        return parseLsof(output)
    }
}

lsof 输出解析
lsof -nP 输出的 NAME 列格式为：

// TCP: 192.168.1.5:52341->142.250.80.46:443
// UDP: *:5353
// 解析 Regex:
let pattern = #"(\S+):(\d+)->(\S+):(\d+)"#
// Group 1: localIP  Group 2: localPort
// Group 3: remoteIP Group 4: remotePort

// 过滤私有 IP:
func isPrivateIP(_ ip: String) -> Bool {
    ip.hasPrefix("127.") || ip.hasPrefix("::1") ||
    ip.hasPrefix("10.")  || ip.hasPrefix("192.168.") ||
    (ip.hasPrefix("172.") && /* 172.16-31 check */)
}

class ArcOverlay: NSObject, MKOverlay {

    let boundingMapRect: MKMapRect
    let start: CLLocationCoordinate2D
    let end: CLLocationCoordinate2D

class ArcOverlayRenderer: MKOverlayRenderer {
    override func draw(_ mapRect: MKMapRect, zoomScale: MKZoomScale, in ctx: CGContext) {
        guard let arc = overlay as? ArcOverlay else { return }
        let startPt = point(for: MKMapPoint(arc.start))
        let endPt   = point(for: MKMapPoint(arc.end))
        // 控制点：中点向上偏移（纬度方向 + 经度跨度 * 0.3）
        let ctrlPt  = controlPoint(start: startPt, end: endPt)
        let path = CGMutablePath()
        path.move(to: startPt)
        path.addQuadCurve(to: endPt, control: ctrlPt)
        ctx.addPath(path)
        ctx.setStrokeColor(arc.color.withAlphaComponent(0.8).cgColor)
        ctx.setLineWidth(1.5 / zoomScale)
        ctx.strokePath()
    }
}

8.6 TracerouteRunner 实现要点
class TracerouteRunner {
    private var process: Process?

    func run(target: String) -> AsyncStream<TracerouteHop> {
        AsyncStream { continuation in
            self.process?.terminate()
            let p = Process()
            p.executableURL = URL(fileURLWithPath: "/usr/sbin/traceroute")
            p.arguments = ["-n", "-m", "20", "-q", "1", target]
            let pipe = Pipe()
            p.standardOutput = pipe
            var hopNumber = 0
            pipe.fileHandleForReading.readabilityHandler = { handle in
                let data = handle.availableData
                guard !data.isEmpty, let line = String(data: data, encoding: .utf8) else {
                    continuation.finish(); return
                }
                hopNumber += 1
                if let hop = Self.parseLine(line, hopNum: hopNumber) {
                    continuation.yield(hop)
                }
            }
            try? p.run()
            self.process = p
        }
    }

    static func parseLine(_ line: String, hopNum: Int) -> TracerouteHop? {
        // Pattern: " 3  142.251.49.1  14.234 ms"

        let hopRegex = #"^\s*(\d+)\s+(\S+)\s+([\d.]+) ms"#

}

8.7 ConnectionStore 增量更新
每秒 lsof 轮询会产生全量连接列表，需要与上一次状态做 diff，避免不必要的 UI 刷新和地图重绘：

@MainActor
class ConnectionStore: ObservableObject {
    @Published var connections: [Connection] = []
    private var connectionMap: [String: Connection] = [:]   // id -> Connection

    func update(with fresh: [Connection]) {
        let freshMap = Dictionary(uniqueKeysWithValues: fresh.map { ($0.id, $0) })
        let added   = fresh.filter { connectionMap[$0.id] == nil }
        let removed = connectionMap.values.filter { freshMap[$0.id] == nil }
        let updated = fresh.filter { c in connectionMap[c.id] != nil }
        connectionMap = freshMap
        // 只通知变更的部分，而非全量替换
        if !added.isEmpty || !removed.isEmpty {
            connections = Array(freshMap.values)
                .sorted { $0.processName < $1.processName }
            mapDelegate?.addOverlays(for: added)
            mapDelegate?.removeOverlays(for: removed)
        }
    }
}

12. 分发策略
12.1 Phase 1 分发
	•	以 .app bundle 形式直接分发，不上架 Mac App Store
	•	提供 Homebrew Cask 安装（参考 spk 的已有 tap 经验）
	•	代码签名：使用 Developer ID Application 证书 + Notarization（需 Apple Developer Program）
	•	如早期测试阶段，可先用 ad-hoc 签名，提示用户执行 xattr -cr /Applications/NetScope.app

12.2 GeoLite2 数据库分发问题
限制
MaxMind GeoLite2 数据库不允许随 App 一起分发，用户必须自行注册免费账号后下载。

	•	首次启动显示 Setup 引导页：步骤 1 - 注册 MaxMind 账号 / 步骤 2 - 下载 GeoLite2-City.mmdb / 步骤 3 - 拖拽到 App 内指定位置
	•	提供「Open MaxMind Download Page」按钮直接跳转
	•	存储路径：~/Library/Application Support/NetScope/GeoLite2-City.mmdb（用户可替换更新）
	•	Preferences 中显示数据库版本和构建日期（读取 mmdb metadata）

12.3 更新策略
	•	集成 Sparkle 框架，提供自动更新检测（appcast XML 托管在 GitHub Releases）
	•	每次发布在 GitHub 创建 Release，上传签名后的 .dmg 或 .zip
	•	Homebrew Cask 配方随版本更新 SHA256 和 URL
